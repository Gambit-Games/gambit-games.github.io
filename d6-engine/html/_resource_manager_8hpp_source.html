80 template<typename T>
81 uint64_t LoadResourceFromMetadata(const std::filesystem::path& filepath)
82 {
83 Resources::Resource<void*> resourceToLoad;
84 Metadata data(filepath.string());
85 resourceToLoad.uuid = data.GetUInt64_T("UUID");
86 if (!resourceToLoad.uuid)
87 {
88 log.Warning("Invalid UUID detected while loading '%s'. A new UUID has been generated.",
89 filepath.string().c_str());
90 resourceToLoad.uuid = UUID();
91 data.SetUInt64_T(resourceToLoad.uuid, "UUID");
92 }
93 resourceToLoad.name = data.GetString("Name");
94 resourceToLoad.filepath = data.GetString("FilePath");
95 resourceToLoad.resourceType = data.GetResourceType(filepath);
96 resourceToLoad.resource = new T(resourceToLoad.filepath.string(), filepath.string());
97
98 all_resources.emplace_back(resourceToLoad);
99
100 return resourceToLoad.uuid;
101 }
