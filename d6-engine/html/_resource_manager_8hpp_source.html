60 template<typename T>
61 static uint64_t LoadResourceFromFile(const std::filesystem::path& filepath)
62 {
63 //@Cecil: Check if file got Metadata, if have, load from Metadata instead
64 if (Metadata::FileExist(filepath.string() + ".d6meta"))
65 {
66 return LoadResourceFromMetadata<T>(filepath.string() + ".d6meta");
67 }
68 else
69 {
70 //normal resource loading
71
72 Resource<void*> resourceToLoad;
73 resourceToLoad.uuid = Random::GetUniqueID();
74 resourceToLoad.name = filepath.stem().string();
75 resourceToLoad.filepath = filepath;
76 resourceToLoad.resource = new T(filepath.string());
77 resourceToLoad.resourcetype = Metadata::GetResourceType(filepath);
78
79 //Metadata Serialization here
80 Metadata::CreateFile(filepath.string());
81 Metadata::LoadData(filepath.string() + ".d6meta");
82 Metadata::SetUInt64_T(resourceToLoad.uuid, "UUID", filepath.string() + ".d6meta");
83 Metadata::SetString(resourceToLoad.name, "Name", filepath.string() + ".d6meta");
84 Metadata::SetString(resourceToLoad.filepath.string(), "FilePath", filepath.string() + ".d6meta");
85 Metadata::SetString(resourceToLoad.filepath.extension().string(), "Extension", filepath.string() + ".d6meta");
86 Metadata::SetType(resourceToLoad.resourcetype, "Type", filepath.string() + ".d6meta");
87
88 all_resources.emplace_back(resourceToLoad);
89
90 return resourceToLoad.uuid;
91 }
92 }
